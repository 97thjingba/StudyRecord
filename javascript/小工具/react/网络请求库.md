export  default  class  BaseApi {

get({ ...params }) {

params.method = 'GET';

// 因为使用 'content-type': 'application/json' 的 header 会导致 400 报错

// 而 Taro 会自动添加，所以给一个空格规避错误

if (!params.header) {

params.header = {

'content-type':  ' '

};

}

return  this._request(params);

}

  

post({ ...params }) {

params.method = 'POST';

if (!params.header) {

params.header = HEADER;

}

return  this._request(params);

}

  

put({ ...params }) {

params.method = 'PUT';

if (!params.header) {

params.header = HEADER;

}

return  this._request(params);

}

  
  

_request({ ...params }) {

const  url = this._generateUrl(params.path);

params.url = url;

console.log("params", params);

return  RequestModel(params).then(res  => {

const { data } = res;

const { ok } = data;

if (data && ok) {

const { result } = data;

return  result;

} else {

const { status } = res;

throw  this._handleError(url, data, status);

}

}).catch((error) => {

throw  error;

});

}

  

_generateUrl = (path) => {

return  DOMAIN + path;

};

  

// 仅处理与服务器商定的错误格式

_handleError = (url, response, status) => {

if (response.errors) {

if (response.errors.length <= 0) {

return  ApiErrorFactory.makeServerReturnWrongError('服务器返回了errors，但errors为空');

}

const  errorObject = response.errors[0];

const  errorMessage = `${errorObject.message}`;

return  ApiErrorFactory.makeNormalError({

ok:  false,

url,

error_type:  'form_validate_error',

message:  errorMessage,

status,

});

}

return  ApiErrorFactory.makeNormalError({

ok:  response.ok,

url,

error_type:  response.error_type,

message:  response.message || response.error_message,

status,

});

};
